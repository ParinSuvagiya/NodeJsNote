   #setter,deleter,property
   class emp:
      def __init__(self,name,worktype):
         self.name=name
         self.worktype=worktype
         #email=f'{name}@aone.com'
   
      def detail(self):
         return self.name,self.worktype
   
      @property
      #property function call without () brekat
      def email(self):
         if self.name==None:
            return 'Not exists'
            return f'{self.name}@aone.com'
   
      @email.setter
      def email(self,str):
         self.name=str.split('@')[0]
   
      @email.deleter
      def email(self):
         self.name=None
         self.worktype=None
   
   e1=emp('bob','fulltime')
   print(e1.detail())

   """

In Python 3.2+ there is an lru_cache decorator which allows us to quickly cache and uncache the return values of a function.
